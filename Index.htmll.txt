<script>
let coins = parseInt(localStorage.getItem('coins')) || 0;

function updateCoins() {
    localStorage.setItem('coins', coins);
}

// Home bo‘limi
function showHome() {
    document.getElementById("content").innerHTML = `
        <h2 class="title">Home</h2>
        <p class="coins-text"><span id="coinsSpan">${coins}</span></p>
        <img src="https://i.ibb.co/2yLZ0d7/home-icon.png" id="coinImg" class="coin-img" alt="tanga">
    `;
    document.getElementById("coinImg").onclick = () => {
        coins += 1;
        updateCoins();
        document.getElementById("coinsSpan").textContent = coins;
    }
}

// Missions bo‘limi
function showMissions() {
    document.getElementById("content").innerHTML = `
        <h2 class="title">Missions</h2>
        <img src="https://i.ibb.co/fYnTq1Y/mission-icon.png" class="coin-img" alt="mission">
        <button id="watchAd" class="ad-button">Reklama ko‘rish</button>
    `;
    document.getElementById("watchAd").onclick = () => {
        alert("Reklama ko‘rildi! Siz 150 tanga oldingiz.");
        coins += 150;
        updateCoins();
    }
}

// Profile bo‘limi
function showProfile() {
    document.getElementById("content").innerHTML = `
        <h2 class="title">Profile</h2>
        <p class="coins-text">Profil bo‘limi</p>
    `;
}

// Pastki nav tugmalari
document.getElementById("home").onclick = () => { setActive("home"); showHome(); };
document.getElementById("missions").onclick = () => { setActive("missions"); showMissions(); };
document.getElementById("profile").onclick = () => { setActive("profile"); showProfile(); };

function setActive(activeId) {
    ["home", "missions", "profile"].forEach(id => document.getElementById(id).classList.remove("active"));
    document.getElementById(activeId).classList.add("active");
}

// Dastlab Home ochiladi
showHome();
</script>
